/* check against previous primes */
bool is_prime(int n, int *primes)
{
	/* only check up to sqrt(n) */
	int max_test = (int)sqrt(n);
	for (int i = 0; primes[i] <= max_test; ++i)
		if (n % primes[i] == 0)
			return false;
	return true;
}
